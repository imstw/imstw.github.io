By 葳 20 January 2023

Columns & Rows Count

–取列(Row)數

Dim RowCount As Long  
RowCount = Worksheets("原始資料表").Range("B" & Rows.Count).End(xlUp).Row

–取欄(Column)數

Dim ColCount As Long  
ColCount = Worksheets("原始資料表").Cells(1, Columns.Count).End(xlToLeft).Column

–取得原始資料範圍

Dim SourceRange As String  
SourceRange = "原始資料表!R1C1:R" & RowCount & "C" & ColCount
